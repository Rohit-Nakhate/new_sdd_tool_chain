398 static inline boolean DoIP_IsFunctionalAddressing(uint16 TargetAddr)
399 {
400  boolean bRetVal = FALSE;
401 
402  // ISO/SAE reserved functional group addresses
403  if ( (TargetAddr >= DOIP_FUNC_ADDR_RANGE_START) && (TargetAddr <= DOIP_FUNC_ADDR_RANGE_END) )
404  {
405  bRetVal = TRUE;
406  }
407 
408  // manufacturer-defined functional group logical addresses
409  else if ( (TargetAddr >= DOIP_FUNC_ADDR_MANUFACTURER_RANGE_START) && (TargetAddr <= DOIP_FUNC_ADDR_MANUFACTURER_RANGE_END) )
410  {
411  bRetVal = TRUE;
412  }
413 
414  // ISO 27145 WWH-OBD functional group address
415  else if ( TargetAddr == DOIP_FUNC_ADDR_WWH_OBD )
416  {
417  bRetVal = TRUE;
418  }
419  else
420  {
421  /* nothing to do */
422  }
423 
424  return bRetVal;
425 }
