171  }
172  break;
173 
174  case DOIP_PL_TYPE_ROUTE_ACT_REQ:
175  if ( DoIP_IsTcpTester(SocketHdl) != FALSE )
176  {
177  /*Receive routing activation request from E.N. */
178  DoIP_Client_RoutingActivation(SocketHdl, RemoteAddrPtr, BufPtr, Length);
179  }

182  case DOIP_PL_TYPE_ALIVE_REQ:
183  (void)DoIP_SendAliveCheckRes(SocketHdl);
184  break;
185 
186  case DOIP_PL_TYPE_ENTITY_STATUS_REQ:
187  (void)DoIP_SendEntityStatusRes();
188  break;

191  (void)DoIP_SendPowerModeRes();
192  break;

194  case DOIP_PL_TYPE_COM_PARAMS_REQ:
195  {
196  uint16 testerIdx = GetTesterID(SocketHdl);
197  if (testerIdx == INVALID_TESTER)
198  {
199  return E_NOT_OK;

217  /* Nothing to do - Connection not activated */
218 
219  }

259  break;
260  case DOIP_PL_TYPE_DIAG_MSG:
261  {
262  uint16 testerIdx = GetTesterID(SocketHdl);
263  if (testerIdx == INVALID_TESTER)

265  return E_NOT_OK;

271  {
272  uint16 SrcAddr = DoIP_Get_UInt16( &DoipPayloadPtr[DOIP_DIAG_MSG_B0_SOURCE_ADDR_MSB]);
273  uint16 TargetAddr = DoIP_Get_UInt16( &DoipPayloadPtr[DOIP_DIAG_MSG_B2_TARGET_ADDR_MSB]);
274  DoIP_PduR_tenTesterList ClientType;
275 
276  /* Check client type (Onboard/Offboard tester) */
277  if (DoIP_IsOffboardTester(SrcAddr) != FALSE)
278  {
279  ClientType = DOIP_PDUR_OffBoardTester;
280  }
281  else if (DoIP_IsOnboardTester(SrcAddr) != FALSE )
282  {
283  ClientType = DOIP_PDUR_OnBoardTester;
284  }
285  else
286  {
287  /* Send according NACK*/
288  (void)DoIP_SendDiagMsgNack(SocketHdl, TargetAddr, SrcAddr, DOIP_MSG_NACK_INVALID_SRC, NULL, 0);
289  DoIP_SetError(DOIP_ERR_INVALID_TESTER);
290  return E_NOT_OK;
291  }
292 
293  /* Check the message length of the DoIP diag. request */
294  if ( (Length > DOIP_MAX_DOIP_TP_MSG_LENGTH) || (Length <= DOIP_DIAG_MSG_TOTAL_HEADER_SIZE) )
295  {
296  (void)DoIP_SendDiagMsgNack(SocketHdl, TargetAddr, SrcAddr, DOIP_MSG_NACK_TOO_LARGE, NULL, 0);
297  DoIP_SetError(DOIP_ERR_DOIP_MSG_LENGTH);
298  return E_NOT_OK;
299  }
300  /* Forward the request to the according target */
301  (void)DoIP_Server_ForwardRequest(SocketHdl, SrcAddr, TargetAddr, SoMgr_SocketTable[SocketHdl].pRxWin, Length, ClientType);
302  }
303  else
304  {
305  /* Tester not in the accordding state*/
306  retVal = E_NOT_OK;
307  }
308 
309  }
310  else
311  {
312  /* Socket is not a TCP one*/
313  retVal = E_NOT_OK;
314  }
315  break;
316  }
317  case DOIP_PL_TYPE_GEN_HEADER_NACK:
318  LOG(LOG_MODULE_doip, LOG_LEVEL_warn, "MSG: GEN_HEADER_NACK (SocketHdl: %d): %>\r\n", SocketHdl, RemoteAddrPtr );
319  break;

321  /* Payload types not supported or unexpected */
322  case DOIP_PL_TYPE_VEHICLE_ID_RESP_ANN:
323  case DOIP_PL_TYPE_ROUTE_ACT_RESP:
324  case DOIP_PL_TYPE_ALIVE_RESP:

346 {
347 
348  SWATT_BYPASS(m_DoIP_Server_GetEcuType_StaticRouting, m_DoIP_Server_GetEcuType_StaticRouting(LogAddr));
349 
350  DOIP_ROUTING_BUS_TYPE routingBus = DOIP_ROUTING_BUS_INVALID;
351 
352  if ( LogAddr == DOIP_LOG_ADDR_GW )
353  {
354  routingBus = DOIP_ROUTING_BUS_GW;
355  }
356  else if ( TRUE == DoIP_IsFunctionalAddressing(LogAddr) )

375 static Std_ReturnType DoIP_Server_ForwardRequest(SoMgr_SocketHdlType socketHdl, uint16 sourceAddr, uint16 targetAddr, struct pbuf * pDoIpDiagReq, uint16 length, DoIP_PduR_tenTesterList ClientType)
376 {
377  Std_ReturnType retVal = E_NOT_OK;
378  uint16 payloadLength = 0u;
379  DOIP_ROUTING_BUS_TYPE ecuType = DoIP_Server_GetEcuType_StaticRouting(targetAddr);
380 
381  payloadLength = length - DOIP_DIAG_MSG_TOTAL_HEADER_SIZE;
382 
383  if (ecuType != DOIP_ROUTING_BUS_DK2F)
384  {
385  /* Remove DoIP header (not for LIN-targets as they need the addresses in DoIP header) */
386  (void)pbuf_free_front( pDoIpDiagReq, DOIP_DIAG_MSG_TOTAL_HEADER_SIZE );
387  }
388 
389 
390  switch (ecuType)
391  {
392  case DOIP_ROUTING_BUS_GW:
393  {
394  NotifResultType diagAckNack_DCM;
395 
396  /* ACKs/ NACKs will be handled inside this function*/
397  diagAckNack_DCM = DoIP_DcmHandleDiagReq(socketHdl, pDoIpDiagReq, sourceAddr, targetAddr, ClientType);
398  if (diagAckNack_DCM == NTFRSLT_OK )
399  {
400  retVal = E_OK;
401  }
402  break;
403  }
404  case DOIP_ROUTING_BUS_FUNCTIONAL:
405  {
406  if ( (payloadLength > 0u) && (payloadLength < DOIP_FUNCTIONAL_MSG_MAX_LENGTH) )
407  {
408  DOIP_RETURN_TYPE diagAckNack_TP;
409  NotifResultType diagAckNack_DCM;
410 
411  /* Pass data to DCM module and transmit functional request to all CANs*/
412  diagAckNack_DCM = DoIP_DcmHandleDiagReq(socketHdl, pDoIpDiagReq, sourceAddr, targetAddr,ClientType);
413  diagAckNack_TP = DoIP_TP_TransmitFunctionalRequest(pDoIpDiagReq, payloadLength, ClientType);
414 
415  DoIP_EcuMonitor.FuncAddrReqCount++;
416 
417  if ((diagAckNack_DCM != NTFRSLT_OK )||(diagAckNack_TP != DOIP_ROUTING_OK))
418  {
419  /* Memory management is still in our hand (ETH task) */
420  /* The NACK from DCM shall be forwarded (already forwarded )*/
421  }
422  else
423  {
424  retVal = E_OK;
425  }
426  }
427  else
428  {
429  (void)DoIP_SendDiagMsgNack(socketHdl, targetAddr, sourceAddr, DOIP_MSG_NACK_TOO_LARGE, NULL, 0);
430  DoIP_SetError(DOIP_ERR_DOIP_MSG_LENGTH);
431  }
432  break;
433  }
434 
435  case DOIP_ROUTING_BUS_DK2F:
436  {
437  DIAGR_RETURN_TYPE diagRet = DIAGR_ROUTING_NOT_OK;
438 
439  /* Forward DoIP request to DiagMasterSlave*/
440  diagRet = DK2F_HandleDoIPRequest(pDoIpDiagReq, payloadLength);
441  /* Memory management is taken over by DiagMasterSlave*/
442  SoMgr_SocketTable[socketHdl].pRxWin = NULL;
443  if (diagRet == DIAGR_ROUTING_OK)
444  {
445  (void)DoIP_SendDiagMsgAck(socketHdl, targetAddr, sourceAddr, DOIP_DIAG_MSG_CONF_ACK, NULL, 0);
446  retVal = E_OK;
447  }
448  else if (diagRet == DIAGR_TARGET_UNREACHABLE)
449  {
450  /* The NACK is set if the target address is unreacheable */
451  (void)DoIP_SendDiagMsgNack(socketHdl, targetAddr, sourceAddr, DOIP_MSG_NACK_UNREACHABLE, NULL, 0);
452  }
453  else
454  {
455  /* The NACK is set if transmission request is not possible due to other reasons (including electric failures */
456  (void)DoIP_SendDiagMsgNack(socketHdl, targetAddr, sourceAddr, DOIP_MSG_NACK_TP_ERROR, NULL, 0);
457  }
458  break;
459  }
460 
461  case DOIP_ROUTING_BUS_CANTP:
462  case DOIP_ROUTING_BUS_CANFD: /* SWATT-coverage-line DOIP_ROUTING_BUS_CANFD is considered the same with DOIP_ROUTING_BUS_CANTP */
463  {
464  DIAGR_RETURN_TYPE diagRet = DIAGR_ROUTING_NOT_OK;
465 
466  diagRet = DynTpDoIP_PduR_Transmit(targetAddr, pDoIpDiagReq, (uint8)ClientType);
467  /* Memory management is taken over by DynTpDoIP */
468  SoMgr_SocketTable[socketHdl].pRxWin = NULL;
469  if (diagRet == DIAGR_ROUTING_OK)
470  {
471  (void)DoIP_SendDiagMsgAck(socketHdl, targetAddr, sourceAddr, DOIP_DIAG_MSG_CONF_ACK, NULL, 0);
472  retVal = E_OK;
473  }
474  else if (diagRet == DIAGR_UNKNOWN_TARGET)
475  {
476  /* The NACK is send if the logical address could not be derrived in a CanID*/
477  (void)DoIP_SendDiagMsgNack(socketHdl, targetAddr, sourceAddr, DOIP_MSG_NACK_UNKNOWN_TARGET, NULL, 0);
478  DoIP_SetError(DOIP_ERR_TP_TX_FAILED);
479  }
480  else
481  {
482  /* The NACK is set if transmission request is not possible due to other reasons (including electric failures */
483  (void)DoIP_SendDiagMsgNack(socketHdl, targetAddr, sourceAddr, DOIP_MSG_NACK_TP_ERROR, NULL, 0);
484  DoIP_SetError(DOIP_ERR_TP_TX_FAILED);
485  }
486  break;
487  }
488  default:
489  {
490  /* Nothing to do */
491  break;
492 
493  }
494 
495  }
496  return retVal;
497 }
