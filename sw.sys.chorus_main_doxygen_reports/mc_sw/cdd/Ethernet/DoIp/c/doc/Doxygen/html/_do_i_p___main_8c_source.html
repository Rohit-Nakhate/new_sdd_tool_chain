234 void DoIP_MainFunction(void)
235 {
236  /* ----- send Vehicle Announcement (unsolicited on session start) */
237  if (DoIP_Status.State == DOIP_STATE_ACTIVATED)
238  {
239  /* A_DoIP_Announce_Wait */
240  if(DoIP_Status.VehAnnTxCtrl.VehAnnounceEnabled == FALSE)
241  {
242  if(DoIP_Status.VehAnnTxCtrl.VehAnnouncePeriod_ticks > 0u)
243  {
244  DoIP_Status.VehAnnTxCtrl.VehAnnouncePeriod_ticks--;
245  if(DoIP_Status.VehAnnTxCtrl.VehAnnouncePeriod_ticks == 0u)
246  {
247  DoIP_Status.VehAnnTxCtrl.VehAnnounceEnabled = TRUE;
248  }
249  }
250  }
251 
252  /* A_DoIP_Announce_Interval */
253  if(DoIP_Status.VehAnnTxCtrl.VehAnnounceEnabled != FALSE)
254  {
255  if(DoIP_Status.VehAnnTxCtrl.VehAnnouncePeriod_ticks == 0u)
256  {
257  /* Check number of Vehicle announcement messages that was send until now */
258  if (DoIP_Status.VehAnnTxCtrl.VehAnnounceTxCnt > 0u)
259  {
260  if ( E_OK == DoIP_SendVehicleIdentificatonRes() )
261  {
262  DoIP_Status.VehAnnTxCtrl.VehAnnounceTxCnt--;
263  }
264  /* Reste interval timer*/
265  DoIP_Status.VehAnnTxCtrl.VehAnnouncePeriod_ticks = (uint16)DOIP_ANNOUNCE_INTERVAL_TICKS;
266  }
267  else
268  {
269  DoIP_Status.VehAnnTxCtrl.VehAnnounceEnabled = FALSE;
270  DoIP_Status.VehAnnTxCtrl.VehAnnounceTxCnt = DOIP_VEHICLE_ID_RES_MSG_COUNT;
271  }
272  }
273  else
274  {
275  DoIP_Status.VehAnnTxCtrl.VehAnnouncePeriod_ticks--;
276  }
277  }
278  }
279 
280  /* ----- Transfer 'CAN tester active' flag to Edge Node
281  * VWMEB-41676: DoIP_MainFunction moved to 10 ms section from EthTask because EdgeNode should have s_CANTester_Active information

283  if (DoIP_Status.CANTester.DiagCANReqRecv != FALSE)

302  DoIP_Client_MainFunction();

331 uint16 DoIP_RxIndication_DecodeNextMessage(SoMgr_SocketHdlType SocketHdl, TcpIp_SockAddrType * RemoteAddrPtr, uint8 * BufPtr, uint16 Length)
332 /* ANALYSIS_REPORT_JUSTIFICATION (2021/22/03 : uidn1598) !-->
333  TOOL_NUMBER(CODECHECK:S44) GUIDELINE(ICR_201804:RULE:C19) The cyclomatic complexity is not too high, it's acceptable to keep it this way.
334  <--! */
335 {
336  uint16 consumedBytes = Length;
337  uint16 bytesToEnqueue = Length;
338 
339  if (SocketHdl >= (SoMgr_SocketHdlType)SOMGR_MAX_NUM_SOCKETS)
340  {
341  DoIP_SetError(DOIP_ERR_INVALID_SOCKET_ID);
342  return Length;
343  }
344 
345  // ensure that max. one DoIP-message is enqueued in RxWindow of receiving socket
346  if (SoMgr_SocketTable[SocketHdl].RemainingMsgLength == 0u)
347  {
348  uint16 PreviousLength = 0u;
349 
350  if ( SoMgr_SocketTable[SocketHdl].pRxWin != NULL)
351  {
352  // there are already some bytes received in previous segment (but less then DOIP_GEN_HEADER_SIZE)
353  PreviousLength = SoMgr_SocketTable[SocketHdl].pRxWin->tot_len;
354  if (PreviousLength >= DOIP_GEN_HEADER_SIZE)
355  {
356  PreviousLength = DOIP_GEN_HEADER_SIZE;
357  DoIP_SetError(DOIP_ERR_GENERAL);
358  }
359  }
360 
361  if ( (PreviousLength + Length) >= DOIP_GEN_HEADER_SIZE )
362  {
363  // at least 'generic DoIP header' has already been received so capture message length
364  uint8* pMsgGenHeader = BufPtr;
365  uint8 msgGenHeader[DOIP_GEN_HEADER_SIZE];
366  uint32 msgLen;
367 
368  if (PreviousLength > 0u)
369  {
370  // DoIP generic header is splitted in previous and current segment
371  (void)memcpy(msgGenHeader, SoMgr_SocketTable[SocketHdl].pRxWin->payload, PreviousLength); // previous segment
372  (void)memcpy(&msgGenHeader[PreviousLength], BufPtr, ((uint32)DOIP_GEN_HEADER_SIZE- PreviousLength)); // current segment
373  pMsgGenHeader = msgGenHeader;
374  }
375 
376  msgLen = DoIP_Get_UInt32( &pMsgGenHeader[DOIP_GEN_HEADER_B4_LENGTH_3_MSB] ) + DOIP_GEN_HEADER_SIZE;
377 
378  if ( Length > msgLen )
379  {
380  bytesToEnqueue = (uint16)msgLen;
381  }
382  }
383 
384  }
385  else
386  {
387  if (SoMgr_SocketTable[SocketHdl].RemainingMsgLength < Length)
388  {
389  bytesToEnqueue = (uint16)SoMgr_SocketTable[SocketHdl].RemainingMsgLength;
390  }
391  }
392 
393  // store received data rxWindow of Socket
394  if ( E_OK != SoMgr_EnqueueRxWindow(SocketHdl, BufPtr, bytesToEnqueue) )
395  {
396  // Message could not be buffered
397  SoMgr_FreeRxWindow(SocketHdl);
398  (void)DoIP_SendDoipHeaderNack(SocketHdl, DOIP_NACK_CODE_OUT_OF_MEMORY);
399  DoIP_SetError(DOIP_ERR_OUT_OF_MEMORY);
400  return consumedBytes;
401  }
402 
403  if (SoMgr_SocketTable[SocketHdl].RemainingMsgLength == 0u)
404  {
405  // start of new DoIP message or we have already received an incomplete header
406 
407  uint16 DoipVersion;
408  uint16 PayloadType;
409  uint32 MessageLength;
410  uint8 *DoipHeaderPtr;
411 
412  if ( SoMgr_SocketTable[SocketHdl].pRxWin->tot_len < DOIP_GEN_HEADER_SIZE )
413  {
414  // DoIP generic header still uncompleted
415  LOG(LOG_MODULE_doip, LOG_LEVEL_debug, "rxIndication: incomplete header from (SocketHdl: %d): %>\r\n", SocketHdl, RemoteAddrPtr );
416  return consumedBytes;
417  }
418 
419  DoipHeaderPtr = SoMgr_SocketTable[SocketHdl].pRxWin->payload;
420 
421  //----- DoIP generic header available
422  DoipVersion = DoIP_Get_UInt16( &DoipHeaderPtr[DOIP_GEN_HEADER_B0_PR_VERSION]);
423  PayloadType = DoIP_Get_UInt16( &DoipHeaderPtr[DOIP_GEN_HEADER_B2_TYPE_MSB]);
424  MessageLength = DoIP_Get_UInt32( &DoipHeaderPtr[DOIP_GEN_HEADER_B4_LENGTH_3_MSB] ) + DOIP_GEN_HEADER_SIZE;
425 
426  if (DoipVersion != DOIP_PROTOCOL_VERSION)
427  {
428  // check special DoIP-Version for VehicleAnnouncement request (-> ISO 13400-2, 2012)
429  boolean isVehicleAnnouncementReq = FALSE;
430  if ( (PayloadType == DOIP_PL_TYPE_VEHICLE_ID_REQ) ||
431  (PayloadType == DOIP_PL_TYPE_VEHICLE_ID_EID_REQ) ||
432  (PayloadType == DOIP_PL_TYPE_VEHICLE_ID_VIN_REQ) )
433  {
434  isVehicleAnnouncementReq = TRUE;
435  }
436 
437  if ( (DoipVersion != DOIP_PROTOCOL_VERSION_VEH_ANN) || (isVehicleAnnouncementReq != TRUE) )
438  {
439  // TODO: consider to parse the stream for next occurence of start of next message (0x02FD),
440  // but we assume that start of next message will be aligned with a following RxIndication
441  LOG(LOG_MODULE_doip, LOG_LEVEL_warn, "invalid DOIP version 0x%04x\r\n", DoipVersion);
442  DoIP_SetError(DOIP_ERR_INVALID_DOIP_VERSION);
443  SoMgr_SetErrorSock(SocketHdl, SOMGR_ERR_INVALID_DOIP_VERSION);
444  (void)DoIP_SendDoipHeaderNack(SocketHdl, DOIP_NACK_CODE_INCORRECT_PATTERN);
445  // TODO: close socket [DoIP-041]
446  // TODO-klpo66: release pBuf
447  return consumedBytes;
448  }
449  }
450 
451  if (MessageLength > Length)
452  {
453  // DoIP message not completely received yet
454  SoMgr_SocketTable[SocketHdl].RemainingMsgLength = (uint32)MessageLength - (uint32)Length;
455  }
456  else
457  {
458  // DoIP message completely received
459  consumedBytes = (uint16)MessageLength;
460  SoMgr_SocketTable[SocketHdl].RemainingMsgLength = 0;
461  }
462 
463  // wait until message is complete
464  if (SoMgr_SocketTable[SocketHdl].RemainingMsgLength == 0u)
465  {
466  // received data can be processed; they are stored in the rxWindow of Socket (-> SoMgr_SocketTable[SocketHdl].pRxWin)
467  (void)DoIP_Server_RxIndication(SocketHdl, RemoteAddrPtr);
468  SoMgr_FreeRxWindow(SocketHdl);
469  }
470  else
471  {
472  // wait for reception of complete DoIP message before further processing
473  }
474  }
475  else // if (SoMgr_SocketTable[SocketHdl].RemainingMsgLength == 0)
476  {
477  //----- remainder of a previously started DoIP-message
478 
479  // LOG(_doip, _debug, "...: Remaining/Received: %d/%d (SocketHdl:0x%x): %>\r\n",
480  // SoMgr_SocketTable[SocketHdl].RemainingMsgLength, Length, SocketHdl, RemoteAddrPtr );
481 
482  if (SoMgr_SocketTable[SocketHdl].RemainingMsgLength > Length)
483  {
484  // DoIP message not completely received yet
485  SoMgr_SocketTable[SocketHdl].RemainingMsgLength -= Length;
486  }
487  else
488  {
489  // DoIP message completely received
490  consumedBytes = (uint16)SoMgr_SocketTable[SocketHdl].RemainingMsgLength;
491  SoMgr_SocketTable[SocketHdl].RemainingMsgLength = 0;
492  }
493 
494  if ( SoMgr_SocketTable[SocketHdl].RemainingMsgLength == 0u )
495  {
496  // complete DoIP message received (non cut-through mode)
497  (void)DoIP_Server_RxIndication(SocketHdl, RemoteAddrPtr);
498  SoMgr_FreeRxWindow(SocketHdl);
499  }
500 
501  } // if (SoMgr_SocketTable[SocketHdl].RemainingMsgLength == 0)
502 
503  return consumedBytes;
504 }
