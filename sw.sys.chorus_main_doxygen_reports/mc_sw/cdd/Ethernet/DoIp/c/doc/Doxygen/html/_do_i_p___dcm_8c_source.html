178 Std_ReturnType DoIP_DcmTransmit(PduIdType DoIPTxPduId, const PduInfoType * PduInfoPtr)
179 {
180  Std_ReturnType retVal = E_NOT_OK;
181  struct pbuf * pBuf = NULL_PTR;
182  DoIP_PduR_tenTesterList ClientType = DOIP_PDUR_NUMBER_OF_TESTERS;
183 
184  /* Extract the client information based on DoIP Tx PDU ID */
185  if(DoIPTxPduId_4_OffBoardTester == DoIPTxPduId)
186  {
187  ClientType = DOIP_PDUR_OffBoardTester;
188  DcmTxPduId = DcmConf_OffBoard_DoIP_DcmDslConnection_ETH_DcmDslProtocolTx;
189  }
190  else if(DoIPTxPduId_4_OnBoardTester == DoIPTxPduId)
191  {
192  ClientType = DOIP_PDUR_OnBoardTester;
193  DcmTxPduId = DcmConf_OnBoard_DoIP_DcmDslConnection_ETH_DcmDslProtocolTx;
194  }
195  else
196  {
197  ClientType = DOIP_PDUR_NUMBER_OF_TESTERS;
198  }
199 
200 
201  /* Check if the client was found */
202  if(ClientType >= DOIP_PDUR_NUMBER_OF_TESTERS)
203  {
204  /* nothing to do */
205  }
206  else
207  {
208  /* Get the socket ID based on client type */
209  SoMgr_SocketHdlType tcpSocketHdl = DoIP_Client_GetTcpTesterSocketHdl((uint8)ClientType);
210 
211  /* Validate socket ID */
212  if (tcpSocketHdl == DOIP_SOCKET_UNKNOWN)
213  {
214  DoIP_SetError(DOIP_ERR_DCM_TX_INVALID_SOCKET_ID);
215  }
216  else
217  {
218  /* Copy data from DCM to pbuf */
219  pBuf = mem_to_pbuf_cbk( DoIP_MemToPbuf, PduInfoPtr->SduLength, (uint16)(DOIP_GEN_HEADER_SIZE + DOIP_DIAG_HEADERSIZE));
220  /* Indicate to DCM [not] successful reception */
221  if (pBuf != NULL)
222  {
223  /* Get DoIP address based on socket ID */
224  uint16 doIpDestinationAddr = DoIP_Client_GetTcpTesterLogAddrFromSocketHdl(tcpSocketHdl);
225  /* forward Diag-response to DoIP-tester */
226  uint8 *DoipMsgPtr = pBuf->payload;
227  DoIP_MakeGenericHeaderNoDefaultInit( DoipMsgPtr, DOIP_PL_TYPE_DIAG_MSG, DOIP_DIAG_HEADERSIZE + (uint32)PduInfoPtr->SduLength );
228  DoIP_WriteBuf16( &DoipMsgPtr[DOIP_GEN_HEADER_SIZE+DOIP_DIAG_MSG_B0_SOURCE_ADDR_MSB], DOIP_LOG_ADDR_GW );
229  DoIP_WriteBuf16( &DoipMsgPtr[DOIP_GEN_HEADER_SIZE+DOIP_DIAG_MSG_B2_TARGET_ADDR_MSB], doIpDestinationAddr);
230  /* Trigger transmission */
231  retVal = SoMgr_TcpTransmit(tcpSocketHdl, pBuf);
232 
233  Dcm_TpTxConfirmation(DcmTxPduId, NTFRSLT_OK);
234  }
235  else
236  {
237  Dcm_TpTxConfirmation(DcmTxPduId, NTFRSLT_E_NOT_OK);
238  }
239  }
240 
241 
242  if (retVal != E_OK)
243  {
244  DoIP_SetError(DOIP_ERR_DCM_TX_FAILED);
245  }
246  DoIPMonitor_InsertResponse(DOIP_ROUTING_BUS_GW, DOIP_LOG_ADDR_GW, ((E_OK == retVal) ? TRUE : FALSE) );
247  }
248 
249  return retVal;
250 }

252 NotifResultType DoIP_DcmHandleDiagReq(SoMgr_SocketHdlType SocketHdl, struct pbuf *pBuf, uint16 SrcAddr, uint16 TargetAddr, DoIP_PduR_tenTesterList ClientType)
253 {
254  NotifResultType dcmRxIndResult = NTFRSLT_E_NOT_OK;
255  BufReq_ReturnType dcmRxBufState = BUFREQ_E_NOT_OK;
256  PduLengthType rxAvailableBufferSize;
257  PduLengthType rxBufferSizeRemaining;
258  uint8 doIpDiagAckNack = DOIP_MSG_NACK_TP_ERROR;
259  PduIdType dcmPduId = (PduIdType) 0xFFFF; /* Make sure it is out of range */
260 
261  /* Check if the target address is a functional or physical address */
262  if ( TRUE == DoIP_IsFunctionalAddressing(TargetAddr) )
263  {
264  if ( DOIP_PDUR_OffBoardTester == ClientType )
265  {
266  dcmPduId = Dcm_OffBoard_DoIP_DcmDslProtocolRx_ETHfunc;
267  }
268  else
269  {
270  /* Physical address for any other. Normaly here should have only 0x1010 target address at this point. */
271  dcmPduId = Dcm_OnBoard_DoIP_DcmDslProtocolRx_ETHfunc;
272  }
273  }
274  else
275  {
276  /* Physical address for any other. Normaly here should have only 0x1010 target address at this point. */
277  if ( DOIP_PDUR_OffBoardTester == ClientType )
278  {
279  dcmPduId = Dcm_OffBoard_DoIP_DcmDslProtocolRx_ETHphy;
280  }
281  else
282  {
283  /* Physical address for any other. Normaly here should have only 0x1010 target address at this point */
284  dcmPduId = Dcm_OnBoard_DoIP_DcmDslProtocolRx_ETHphy;
285  }
286  }
287 
288  /* Validate DCM pdu id */
289  if(dcmPduId != 0xFFFFu) /* SWATT-coverage-line dcmPduId can't remain 0xFFFF because of the conditions above */
290  {
291  /* Sending data to DCM on the proper channel */
292  uint16 udsLength = pbuf_Size(pBuf);
293  dcmRxBufState = Dcm_StartOfReception(dcmPduId, udsLength, &rxAvailableBufferSize);
294  if (BUFREQ_OK == dcmRxBufState)
295  {
296  if((rxAvailableBufferSize >= udsLength) && (udsLength > 0u))
297  {
298  /* Copy buffered UDS message to DCM storage */
299  PduInfoType dcmRxPduData;
300  for ( ; (pBuf != NULL) && (dcmRxBufState == BUFREQ_OK); pBuf = pBuf->next)
301  {
302  dcmRxPduData.SduDataPtr = pBuf->payload;
303  dcmRxPduData.SduLength = pBuf->len;
304  dcmRxBufState = Dcm_CopyRxData(dcmPduId, &dcmRxPduData, &rxBufferSizeRemaining);
305  }
306  }
307  else
308  {
309  dcmRxBufState = BUFREQ_E_NOT_OK;
310  }
311  }
312  else
313  {
314  /* Nothing to do */
315  }
316  }
317 
318  /* Check for DCM indication result and proper doip ACK/NACK */
319  switch (dcmRxBufState)
320  {
321  case BUFREQ_OK: /* Connection has been accepted and diag. request copied succesfully in DCM buffer */
322  {
323  doIpDiagAckNack = DOIP_DIAG_MSG_CONF_ACK;
324  dcmRxIndResult = NTFRSLT_OK;
325  /* Dcm_CopyRxData successful - inform about active diagnostics */
326  RetriggerSI_DiagnoseAktivTimer();
327  DiagnosticTimersNotifyDiagnoseActiveStatus(TRUE);
328  }
329  break;
330 
331  case BUFREQ_E_NOT_OK: /* Connection has been rejected, reception is aborted */
332  {
333  doIpDiagAckNack = DOIP_MSG_NACK_UNREACHABLE;
334  dcmRxIndResult = NTFRSLT_E_NOT_OK;
335  }
336  break;
337 
338  case BUFREQ_E_OVFL: /* No buffer of the required length can be provided; reception is aborted */
339  {
340  doIpDiagAckNack = DOIP_MSG_NACK_OUT_OF_MEMORY;
341  dcmRxIndResult = NTFRSLT_E_NOT_OK;
342  }
343  break;
344 
345  default:
346  {
347  doIpDiagAckNack = DOIP_MSG_NACK_TP_ERROR;
348  dcmRxIndResult = NTFRSLT_E_NOT_OK;
349  break;
350  }
351  }
352 
353  /* Send ACK/NACK*/
354  if(doIpDiagAckNack == DOIP_DIAG_MSG_CONF_ACK)
355  {
356  (void)DoIP_SendDiagMsgAck(SocketHdl, DOIP_LOG_ADDR_GW, SrcAddr, DOIP_DIAG_MSG_CONF_ACK, NULL, 0);
357  }
358  else
359  {
360  (void)DoIP_SendDiagMsgNack(SocketHdl, DOIP_LOG_ADDR_GW, SrcAddr, doIpDiagAckNack, NULL, 0);
361  DoIP_SetError(DOIP_ERR_DCM_RX_COPY_FAILED);
362  }
363 
364  /*If VWMEB-39742 repeants its self add the call of Dcm_TpRxIndication in the next call of DoIP_DcmMain */
365  Dcm_TpRxIndication(dcmPduId, dcmRxIndResult);
366 
367  return dcmRxIndResult;
368 }
