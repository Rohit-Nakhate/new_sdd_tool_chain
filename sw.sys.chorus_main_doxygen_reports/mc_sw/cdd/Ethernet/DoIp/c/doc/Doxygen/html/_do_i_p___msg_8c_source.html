297 Std_ReturnType DoIP_SendComParamsAck(SoMgr_SocketHdlType SocketHdl)
298 {
299  Std_ReturnType retVal = E_NOT_OK;
300  uint16 resLength;
301  uint16 cnt;
302 
303  if (DoIP_TpComParam.ComParamIdCount > DOIP_NUMBER_OF_COM_PARAMS)
304  {
305  DoIP_TpComParam.ComParamIdCount = DOIP_NUMBER_OF_COM_PARAMS;
306  }
307 
308  resLength = DOIP_COM_PARAM_ADDR_FIELD_LEN + (DoIP_TpComParam.ComParamIdCount * DOIP_COM_PARAM_ACK_FIELD_LEN);
309 
310  struct pbuf * pBuf = pbuf_alloc( DOIP_GEN_HEADER_SIZE + resLength );
311  if (pBuf != NULL_PTR)
312  {
313  uint8 *DoipMsgPtr = pBuf->payload;
314  DoIP_MakeGenericHeader( DoipMsgPtr, DOIP_PL_TYPE_COM_PARAMS_RES, resLength );
315 
316  /* Swap source and target adresses - VWMEB-24421 */
317  /* Source adress - logical addr. of ECU */
318  DoIP_WriteBuf16( &DoipMsgPtr[DOIP_GEN_HEADER_SIZE + DOIP_COM_PARAM_SOURCE_ADDR], DoIP_TpComParam.EcuLogAddr );
319  /* Destination adress - logical client addr. */
320  DoIP_WriteBuf16( &DoipMsgPtr[DOIP_GEN_HEADER_SIZE + DOIP_COM_PARAM_TARGET_ADDR], DoIP_TpComParam.TesterLogAddr );
321 
322  for (cnt = 0; cnt < DoIP_TpComParam.ComParamIdCount; cnt++)
323  {
324  uint16 offset = cnt * DOIP_COM_PARAM_ACK_FIELD_LEN;
325  DoIP_WriteBuf32( &DoipMsgPtr[DOIP_GEN_HEADER_SIZE + DOIP_COM_PARAM_VALUE_OFFSET + offset ], DoIP_TpComParam.Params[cnt].ComParamId );
326  DoIP_WriteBuf16( &DoipMsgPtr[DOIP_GEN_HEADER_SIZE + DOIP_COM_PARAM_VALUE_OFFSET + offset + DOIP_COM_PARAM_VALUE_OFFSET], DoIP_TpComParam.Params[cnt].ComParamAckCode );
327  }
328 
329  pBuf->tot_len = pBuf->len = DOIP_GEN_HEADER_SIZE + resLength;
330  retVal = SoMgr_TcpTransmit(SocketHdl, pBuf);
331  }
332  else
333  {
334  DoIP_SetError(DOIP_ERR_OUT_OF_MEMORY);
335  }
336 
337  return retVal;
338 }
