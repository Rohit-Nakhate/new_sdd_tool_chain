150 void DoIPMonitor_InsertCanID(const TCanRcvCbData* cbd)
151 {
152  uint16 cnt;
153  uint16 maxIdx = DOIP_MIN(DoIP_EcuMonitor.CanIdCount, DOIP_NUMBER_OF_CANIDS);
154 
155  /* cache the last 4 used CanIDs */
156  #define CACHE_DOIP_CANID_SIZE (4u)
157  static uint32 Cache_DoIP_CanIdTable[CACHE_DOIP_CANID_SIZE] = { 0xFFFFFFFFu, 0xFFFFFFFFu, 0xFFFFFFFFu, 0xFFFFFFFFu };
158  static uint8 Cache_DoIP_ComChannelTable[CACHE_DOIP_CANID_SIZE] = { 0xFF, 0xFF, 0xFF, 0xFF};
159  static uint16 Cache_DoIP_LastUsedIdx = 0;
160 
161  for (cnt = 0u; cnt < CACHE_DOIP_CANID_SIZE; cnt++)
162  {
163  if ( (Cache_DoIP_CanIdTable[cnt] == cbd->ID) && (Cache_DoIP_ComChannelTable[cnt] == cbd->ComChannel) )
164  {
165  /* CanId already stored */
166  return;
167  }
168  }
169 
170  /* CanId not in cache, search in 'DoIP_CanIdTable' */
171  for (cnt = 0; cnt < maxIdx; cnt++)
172  {
173  if ( (DoIP_CanIdTable[cnt] == cbd->ID) && (DoIP_ComChannelTable[cnt] == cbd->ComChannel) )
174  {
175  /* CanId already stored, copy to cache to speed up further accesses */
176  Cache_DoIP_LastUsedIdx++;
177  if (Cache_DoIP_LastUsedIdx >= CACHE_DOIP_CANID_SIZE) { Cache_DoIP_LastUsedIdx = 0; } /* SWATT-coverage-line TRUE case can't be done because Cache_DoIP_LastUsedIdx is localy set to 0 */
178  Cache_DoIP_CanIdTable[Cache_DoIP_LastUsedIdx] = cbd->ID;
179  Cache_DoIP_ComChannelTable[Cache_DoIP_LastUsedIdx] = (uint8)cbd->ComChannel;
180  return;
181  }
182  }
183 
184  /* CanId not received so far, so store it */
185  if (DoIP_EcuMonitor.CanIdNextToWrite >= DOIP_NUMBER_OF_ECUS) { DoIP_EcuMonitor.CanIdNextToWrite = 0; }
186  DoIP_CanIdTable[DoIP_EcuMonitor.CanIdNextToWrite] = cbd->ID;
187  DoIP_ComChannelTable[DoIP_EcuMonitor.CanIdNextToWrite] = (uint8)cbd->ComChannel;
188  DoIP_EcuMonitor.CanIdNextToWrite++;
189  DoIP_EcuMonitor.CanIdCount++;
190  return;
191 }

230 uint16 DoIPMonitor_Lookup(uint16 LogAddr)
231 {
232  uint16 cnt;
233  uint16 searchIdx = DoIP_EcuMonitor.LastUsedIndex; // start search with last routed ECU
234  uint16 maxIdx = DOIP_MIN(DoIP_EcuMonitor.EcuCount, DOIP_NUMBER_OF_ECUS);
235 
236  for (cnt = 0; cnt < maxIdx; cnt++)
237  {
238  if (searchIdx >= maxIdx)
239  {
240  searchIdx = 0;
241  }
242  if (DoIP_EcuTable[searchIdx].LogAddr == LogAddr)
243  {
244  DoIP_EcuMonitor.LastUsedIndex = searchIdx; // next time begin search with last used ECU
245  return searchIdx;
246  }
247  searchIdx++;
248  }
249 
250  // LogAddr not routed yet, so reserve new entry
251  DoIP_EcuMonitor.EcuCount++;
252 
253  // critical section protection omitted on purpose because of CPU-load (only debugging data concerned)
254  // SuspendAllInterrupts();
255  searchIdx = DoIP_EcuMonitor.LastUsedIndex;
256  if (DoIP_EcuMonitor.EcuCount != 1U)
257  {
258  searchIdx++;
259  }
260  if (searchIdx >= DOIP_NUMBER_OF_ECUS)
261  {
262  searchIdx = 0;
263  }
264  DoIP_EcuMonitor.LastUsedIndex = searchIdx;
265  // ResumeAllInterrupts();
266 
267  // init new entry
268  (void)memset( &DoIP_EcuTable[searchIdx], 0x0, sizeof(DOIP_ECU_TYPE) );
269  DoIP_EcuTable[searchIdx].LogAddr = LogAddr;
270  return searchIdx;
271 }
