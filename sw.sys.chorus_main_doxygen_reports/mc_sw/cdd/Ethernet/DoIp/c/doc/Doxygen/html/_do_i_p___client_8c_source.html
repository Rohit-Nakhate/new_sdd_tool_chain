394

396  {
397  DoIP_TesterTable[testerIdx].TcpServerSocketHdl = SocketHdl;
398  (void)Eth_CopySockAddr(&DoIP_TesterTable[testerIdx].SockAddrTcp, RemoteAddrPtr);
399 
400  if (DoIP_Status.State == DOIP_STATE_TERMINATED)
401  {
402  // do not accept new TCP-connection - we are on the way to Bootloader ...

435 void DoIP_Client_RoutingActivation(SoMgr_SocketHdlType SocketHdl, TcpIp_SockAddrType * RemoteAddrPtr, const uint8 * BufPtr, uint16 Length)
436 {
437  /* The routing activation request is received from the tester and the 'routing activation response' shall be send now so
438  * the tester logical address is extracted and after send the reply, the tester state shall be set to DOIP_TESTER_STATE_ROUTING*/
439  UNREFERENCED_PARAMETER(Length);
440 
441  uint16 testerIdx = GetTesterID(SocketHdl);
442 
443  const uint8 *DoipPayloadPtr = &BufPtr[DOIP_GEN_HEADER_B8_PAYLOAD]; // start of payload type specific message content
444  uint16 TesterLogAddr = DoIP_Get_UInt16( &DoipPayloadPtr[DOIP_ROU_ACT_RESP_B0_TESTER_ADDR_MSB]);
445 
446  if (testerIdx != (uint16)INVALID_TESTER)
447  {
448  Std_ReturnType res;
449  if ( FALSE == DoIP_IsTester(TesterLogAddr) )
450  {
451  LOG(LOG_MODULE_doip, LOG_LEVEL_warn, "MSG: ROUTE_ACT_REQ invalid Tester-Addr (SocketHdl: %d, %>): 0x%x\r\n", SocketHdl, RemoteAddrPtr, TesterLogAddr);
452  DoIP_SetError(DOIP_ERR_INVALID_TESTER);
453  (void)DoIP_SendRoutingActivationRes(SocketHdl, TesterLogAddr, DOIP_ROU_ACT_RESP_UNKNOWN_SRC);
454  }
455  else
456  {
457  DoIP_TesterTable[testerIdx].TesterLogAddr = TesterLogAddr;
458 
459  if (DoIP_TesterTable[testerIdx].State == DOIP_TESTER_STATE_ACCEPTED)
460  {
461  LOG(LOG_MODULE_doip, LOG_LEVEL_debug, "MSG: ROUTE_ACT_REQ (SocketHdl: %d): %>\r\n", SocketHdl, RemoteAddrPtr );
462  res = DoIP_SendRoutingActivationRes(SocketHdl, TesterLogAddr, DOIP_ROU_ACT_RESP_SUCCESS);
463  if (res == E_OK)
464  {
465  DoIP_TesterTable[testerIdx].State = DOIP_TESTER_STATE_ROUTING;
466  /* Start the timer for handling unrequested uds response (11 01/ 10 01) */
467  DoIP_TesterTable[testerIdx].GeneralTimerTicks = (uint16)DOIP_UDS_RESP_DELAY_TICKS;
468  }
469  else
470  {
471  DoIP_SetError(DOIP_ERR_TX_ERR);
472  }
473  }
474  else
475  {
476  LOG(LOG_MODULE_doip, LOG_LEVEL_warn, "MSG: ROUTE_ACT_REQ in state %s (SocketHdl: %d): %>\r\n",
477  DOIP_getTesterStateStr(DoIP_TesterTable[testerIdx].State), SocketHdl, RemoteAddrPtr );
478  (void)DoIP_SendRoutingActivationRes(SocketHdl, TesterLogAddr, DOIP_ROU_ACT_RESP_WRONG_SRC);
479  DoIP_TesterTable[testerIdx].State = DOIP_TESTER_STATE_IDENTIFIED;
480  }
481  }
482  }
483  else
484  {
485  (void)DoIP_SendRoutingActivationRes(SocketHdl, TesterLogAddr, DOIP_ROU_ACT_RESP_WRONG_SRC);
486  }
487 }

535 void DoIP_Client_MainFunction(void)
536 {
537  uint8 cnt;
538 
539  /* Iterate over tester table */
540  for (cnt = 0; cnt < DOIP_MAX_NUM_TESTERS; cnt++)
541  {
542  /* Check tester state - VWMEB-51775 */
543  if( DoIP_TesterTable[cnt].State == DOIP_TESTER_STATE_ROUTING)
544  {
545  /* Check delay between routing activation response and first unrequested UDS response */
546  if(DoIP_TesterTable[cnt].GeneralTimerTicks > 0u)
547  {
548  DoIP_TesterTable[cnt].GeneralTimerTicks--;
549  if(DoIP_TesterTable[cnt].GeneralTimerTicks == 0u)
550  {
551  /* special use-case: in case there was a previous programming-session (in FBL) Chorus-App has to send the UDS reply */
552  uint16 TesterLogAddr_ProgSession = (uint16)fblGetTesterAddr(); /* tester address from previous programming session */
553  uint16 TesterLogAddr = DoIP_TesterTable[cnt].TesterLogAddr;
554  if (TesterLogAddr_ProgSession == TesterLogAddr)
555  {
556  PduInfoType pduInfo = fblGetDoIPResponse();
557  if ((pduInfo.SduLength > 0u) && (pduInfo.SduLength < 200u))
558  {
559  /* Send doip response */
560  Std_ReturnType retVal = DoIP_SendDiagMsg(DoIP_TesterTable[cnt].TcpServerSocketHdl, DOIP_LOG_ADDR_GW, TesterLogAddr, pduInfo.SduDataPtr, pduInfo.SduLength);
561  if (retVal == E_OK)
562  {
563  LOG(LOG_MODULE_doip, LOG_LEVEL_info, "DoIP: send UDS reply of programming session (Tester 0x%x)\r\n", TesterLogAddr);
564  }
565  else
566  {
567  /* Reply of FBL response failed */
568  LOG(LOG_MODULE_doip, LOG_LEVEL_warn, "DoIP: send UDS reply of programming session FAILED! (Tester 0x%x)\r\n", TesterLogAddr);
569  }
570  }
571  }
572  }
573  }
574  }
