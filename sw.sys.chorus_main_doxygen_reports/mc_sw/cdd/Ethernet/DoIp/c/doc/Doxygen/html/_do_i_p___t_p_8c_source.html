112 DOIP_RETURN_TYPE DoIP_TP_TransmitFunctionalRequest(struct pbuf *pbufPtr, uint16 length, DoIP_PduR_tenTesterList ClientType)
113 {
114  DOIP_RETURN_TYPE diagr_ret = DOIP_ROUTING_OK;
115  Tuint32 FuncCanId, FuncCanId_FD;
116  uint16 byteCnt;
117 
118  static uint8 data[DOIP_FUNCTIONAL_MSG_MAX_LENGTH]; /* Functional Request Data */
119 
120  /* Functional Request has to fit into a Single Frame on all CAN Busses */
121  if (length < DOIP_FUNCTIONAL_MSG_MAX_LENGTH)
122  {
123  /* Prepare Functional ISOTP Data for Tx */
124  data[DOIP_TP_PCI_BYTE0] = (uint8)length; /* Bit 7...4 = 0x0, Bit 3...0 = length */
125  for (byteCnt = DOIP_TP_SF_DATA_BYTE0; byteCnt < DOIP_FUNCTIONAL_MSG_MAX_LENGTH; byteCnt++)
126  {
127  if ( (byteCnt-DOIP_TP_SF_DATA_BYTE0) < length )
128  {
129  /* Fill Data Byte */
130  data[byteCnt] = pbufPtr->payload[byteCnt-1u]; /* Allowed since Memory Block is greater than 8 Bytes! */
131  }
132  else
133  {
134  /* Fill Padding Byte */
135  data[byteCnt] = DOIP_TP_PADDING_BYTE;
136  }
137  }
138 
139  if (ClientType == DOIP_PDUR_OnBoardTester)
140  {
141  FuncCanId = OBDC_Funktionaler_Req_All;
142  FuncCanId_FD = OBDC_Funktionaler_Req_All_FD;
143  }
144  else
145  {
146  FuncCanId = ISO_Funktionaler_Req_All;
147  FuncCanId_FD = ISOx_Funkt_Req_All_FD;
148  }
149 
150  /* Send Data on all CAN Busses */
151  /* Classic CANs */
152  (void)CanTransmitDirect(_MEB_KCAN, FuncCanId, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH);
153  (void)CanTransmitDirect(_MEB_CCAN, FuncCanId, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH); /* B3 */
154  /* (void)CanTransmitDirect(_MEB_AFSCAN, FuncCanId, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH); */ /* B2 */
155 
156  /* FD CANs */
157  (void)gTransmitDirectCtrl(_MEB_ABCANFD, FuncCanId_FD, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH, g_DoIP_FuncFlag);
158  (void)gTransmitDirectCtrl(_MEB_FCANFD, FuncCanId_FD, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH, g_DoIP_FuncFlag);
159  (void)gTransmitDirectCtrl(_MEB_ACANFD, FuncCanId_FD, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH, g_DoIP_FuncFlag);
160  (void)gTransmitDirectCtrl(_MEB_FASCANFD, FuncCanId_FD, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH, g_DoIP_FuncFlag);
161  (void)gTransmitDirectCtrl(_MEB_EVCANFD, FuncCanId_FD, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH, g_DoIP_FuncFlag);
162  (void)gTransmitDirectCtrl(_MEB_FASCANFD2, FuncCanId_FD, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH, g_DoIP_FuncFlag);
163  (void)gTransmitDirectCtrl(_MEB_K2CANFD, FuncCanId_FD, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH, g_DoIP_FuncFlag);
164  (void)gTransmitDirectCtrl(_MEB_MDKCANFD, FuncCanId_FD, (const Tuint8*)&data[0], DOIP_FUNCTIONAL_MSG_MAX_LENGTH, g_DoIP_FuncFlag);
165  }
166  else
167  {
168  /*UDS Message is too long to fit into one single CAN Frame (with TP PCI)*/
169  diagr_ret = DOIP_MSG_TOO_LARGE;
170  }
171  return diagr_ret;
172 }

302 Std_ReturnType DoIP_TP_ComParamRequest(uint8 * BufPtr, uint16 Length)
303 {
304  // note: min 'Length' is already checked by caller
305  DoIP_TpComParam.ComParamIdCount = 0; // reset number of received ComParams
306  DoIP_TpComParam.TesterLogAddr = DoIP_Get_UInt16( &BufPtr[DOIP_COM_PARAM_SOURCE_ADDR]);
307  DoIP_TpComParam.EcuLogAddr = DoIP_Get_UInt16( &BufPtr[DOIP_COM_PARAM_TARGET_ADDR]);
308 
309  BufPtr += DOIP_COM_PARAM_ADDR_FIELD_LEN;
310  Length -= DOIP_COM_PARAM_ADDR_FIELD_LEN;
311 
312  while ( (Length >= DOIP_COM_PARAM_ITEM_FIELD_LEN) && (DoIP_TpComParam.ComParamIdCount < DOIP_NUMBER_OF_COM_PARAMS) )
313  {
314  uint32 comParamId = DoIP_Get_UInt32( &BufPtr[DOIP_COM_PARAM_ID_OFFSET] );
315  uint32 comParamValue = DoIP_Get_UInt32( &BufPtr[DOIP_COM_PARAM_VALUE_OFFSET] );
316 
317  DoIP_TpComParam.Params[DoIP_TpComParam.ComParamIdCount].ComParamId = comParamId;
318  DoIP_TpComParam.Params[DoIP_TpComParam.ComParamIdCount].ComParamValue = comParamValue;
319 
320  switch (comParamId)
321  {
322  case DOIP_COM_PARAM_ID_ST_MIN_OVERRIDE:
323  DynTpDoIP_ForceSTmin_Store(DoIP_TpComParam.EcuLogAddr, (uint8)comParamValue);
324  DoIP_TpComParam.Params[DoIP_TpComParam.ComParamIdCount].ComParamAckCode = DOIP_COM_PARAM_ACK_SUCCESS;
325  break;
326 
327  default:
328  DoIP_TpComParam.Params[DoIP_TpComParam.ComParamIdCount].ComParamAckCode = DOIP_COM_PARAM_ACK_NOT_SUPPORTED;
329  break;
330  }
331 
332  LOG(LOG_MODULE_doip, LOG_LEVEL_info, "DoIP: ComParam (ID: %d, value: 0x%d, Addr: 0x%x)\r\n", comParamId, comParamValue, DoIP_TpComParam.EcuLogAddr);
333 
334  DoIP_TpComParam.ComParamIdCount++;
335  Length -= DOIP_COM_PARAM_ITEM_FIELD_LEN;
336  BufPtr += DOIP_COM_PARAM_ITEM_FIELD_LEN;
337  }
338 
339  return E_OK;
340 }
