322 FUNC(void, SOMGR_CODE) SoMgr_TcpIpEvent
323 (
324  TcpIp_SocketIdType SocketId,
325  TcpIp_EventType Event
326 )
327 {
328  SoMgr_SocketHdlType socketHdl = SoMgr_getSocketHdl(SocketId);
329 
330  if (socketHdl < (SoMgr_SocketHdlType)SOMGR_MAX_NUM_SOCKETS)
331  {
332  LOG(LOG_MODULE_eth, LOG_LEVEL_warn, "EVENT: %s (SocketHdl: %d)\r\n", SoMgr_getEventStr(Event), socketHdl);
333 
334  SoMgr_SocketTable[socketHdl].EventCount++;
335  SoMgr_SocketTable[socketHdl].LastEvent = Event;
336 
337  if ( (Event == TCPIP_TCP_RESET) || (Event == TCPIP_TCP_CLOSED) || (Event == TCPIP_TCP_FIN_RECEIVED) || (Event == TCPIP_UDP_CLOSED) )
338  {
339  // notify correct upper layer (currently it's only DoIP)
340  DoIP_Client_TcpIp_Event(socketHdl, Event);
341 
342  if (SoMgr_SocketTable[socketHdl].SocketId != TCPIP_SOCKETID_INVALID)
343  {
344  TcpIp_SocketIdType closeSocketId = SoMgr_SocketTable[socketHdl].SocketId;
345  SoMgr_ReleaseSocket(socketHdl); // 'SoMgr_SocketTable[].SocketId' is reset, so no 'TcpIp_Close' recursion can happen;
346  if (Event == TCPIP_TCP_FIN_RECEIVED)
347  {
348  // note: on all other 'close' events the connection is already closed and 'TcpIp_Close' should not be called
349  (void)TcpIp_Close(closeSocketId, FALSE /* Abort */);
350  }
351  }
352  }
353  }
354  else
355  {
356  SoMgr_SetError(SOMGR_ERR_INVALID_SOCKET_ID);
357  }
358 
359  return;
360 }
