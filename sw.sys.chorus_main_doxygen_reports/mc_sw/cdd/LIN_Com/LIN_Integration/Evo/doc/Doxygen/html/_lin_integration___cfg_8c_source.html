147 scheduleType LinInt_GetScheduleType(const l_schedule_handle schedule)
148 {
149  scheduleType retVal;
150  // contains all the LIN schedule tables from the LDF files
151  // select the category of the schedule table
152  switch (schedule)
153  {
154  // all flash table from LDF
155  case LIN_TAB_BEM_Flash:
156  case LIN_TAB_SHL_Flash:
157  case LIN_TAB_MFL_Flash:
158  retVal = flash_sched;
159  break;
160  // all main (default) table from LDF
161  case LIN_TAB_BEM_st_main:
162  case LIN_TAB_MFL_st_main:
163  case LIN_TAB_SHL_st_main:
164  case LIN_TAB_DWA_st_main:
165  case LIN_TAB_LIN5_st_main:
166  case LIN_TAB_VIP_st_main:
167  case LIN_TAB_LIN7_st_main:
168  retVal = appl_sched;
169  break;
170  // all diagnose Request table from LDF
171  case LIN_TAB_BEM_DiagRequest:
172  case LIN_TAB_MFL_DiagRequest:
173  case LIN_TAB_SHL_DiagRequest:
174  case LIN_TAB_DWA_DiagRequest:
175  case LIN_TAB_LIN5_DiagRequest:
176  case LIN_TAB_VIP_DiagRequest:
177  case LIN_TAB_LIN7_DiagRequest:
178  retVal = diag_req_sched;
179  break;
180  // all diagnose Response table from LDF
181  case LIN_TAB_BEM_DiagResponse:
182  case LIN_TAB_MFL_DiagResponse:
183  case LIN_TAB_SHL_DiagResponse:
184  case LIN_TAB_DWA_DiagResponse:
185  case LIN_TAB_LIN5_DiagResponse:
186  case LIN_TAB_VIP_DiagResponse:
187  case LIN_TAB_LIN7_DiagResponse:
188  retVal = diag_resp_sched;
189  break;
190  // all alternative table from LDF
191  case LIN_TAB_DWA_DWA_scharf:
192  retVal = altern_sched;
193  break;
194  // tables that should not be requested by components
195  case LIN_TAB_GOTO_SLEEP:
196  case L_NULL_SCHEDULE:
197  case LIN_SCHED_TAB_NR_MAX:
198  default:
199  retVal = invalid_sched;
200  break;
201  }
202  return retVal;
203 }
