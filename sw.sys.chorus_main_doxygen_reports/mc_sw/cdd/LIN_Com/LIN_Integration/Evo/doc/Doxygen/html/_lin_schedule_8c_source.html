590 LIN_LOCAL void LinScheduleDiagnosis(const l_ifc_handle ifc, const uint8 nextScheduledEntry, const scheduleType requestedScheduleType,
591  const scheduleType activatedScheduleType)
592 {
593  SWATT_BYPASS(LinScheduleDiagnosis,m_LinScheduleDiagnosis( ifc, nextScheduledEntry, requestedScheduleType, activatedScheduleType));
594  LinScheduleDataT *const scheduleData = &(LinScheduleData[ifc]);
595 
596  // on schedule table switch
597  if (1u == nextScheduledEntry)
598  {
599  // reenable DTCs
600  scheduleData->suppressCommunicationDTC = FALSE;
601  // reset buildin list status
602  if ((appl_sched == activatedScheduleType) || (altern_sched == activatedScheduleType))
603  {
604  BIL_IstverbauReset(ifc);
605  }
606  }
607 
608  // diag request send, and response table is active
609  if ((FALSE != scheduleData->requestSent) && (diag_resp_sched == requestedScheduleType))
610  {
611  // sending request completed.
612  scheduleData->requestSent = FALSE;
613  // Switch to diag response schedule after 40ms instead of 50ms
614  if ((scheduleData->diagCounter + LIN_SCHEDULE_10MS) <= (uint8)LIN_DIAG_SCHED_TIMER_MAX)
615  {
616  scheduleData->diagCounter += (uint8)LIN_SCHEDULE_10MS;
617  }
618  else
619  {
620  // reset timer
621  scheduleData->diagCounter = (uint8)LIN_DIAG_SCHED_TIMER_MAX;
622  }
623  }
624 }
