709 static Std_ReturnType EDRN_SetDTCBuffer(const uint8 DEMMemBlock, const Dem_DTCStatusMaskType statusMask, Dcm_MsgContextType * pMsgContext, uint16 * const lub_respIndex)
710 {
711  Std_ReturnType lub_retVal = E_OK;
712  uint16 lub_dtcCount = K_DIAG_ZERO_VAL;
713  uint16 lub_index = K_DIAG_ZERO_VAL;
714 
715  Dem_ReturnGetNextFilteredDTCType lub_dtcFilterRes = DEM_FILTERED_NO_MATCHING_DTC;
716  Dem_DTCType luw_currentDTC = K_DIAG_ZERO_VAL;
717  Dem_ReturnSetFilterType lub_filterRes = DEM_WRONG_FILTER;
718  Dem_EventStatusExtendedType lub_statusOfDtc = K_DIAG_ZERO_VAL;
719 
720  /* First, filter all DTC using the provided mask */
721  lub_filterRes = Dem_SetDTCFilter(statusMask,
722  DEM_DTC_KIND_ALL_DTCS,
723  DEM_DTC_FORMAT_UDS,
724  DEMMemBlock,
725  DEM_FILTER_WITH_SEVERITY_NO,
726  DEM_SEVERITY_NO_SEVERITY,
727  DEM_FILTER_FOR_FDC_NO);
728 
729  if ( DEM_FILTER_ACCEPTED == lub_filterRes )
730  {
731  lub_retVal = Dem_GetNumberOfFilteredDTC(&lub_dtcCount);
732 
733  for(lub_index=K_DIAG_ZERO_VAL; lub_index<lub_dtcCount; lub_index++) /* Circle through filtered list to get the DTCs */
734  {
735  lub_dtcFilterRes = Dem_GetNextFilteredDTC(&luw_currentDTC, &lub_statusOfDtc);
736  luw_currentDTC = luw_currentDTC << 8;
737 
738  if ( DEM_FILTERED_OK == lub_dtcFilterRes )
739  {
740  (void)memcpy(&pMsgContext->resData[*lub_respIndex], &luw_currentDTC, SIZE_OF_DTC);
741  *lub_respIndex += SIZE_OF_DTC;
742 
743  pMsgContext->resData[(*lub_respIndex)++] = (Dcm_MsgItemType)lub_statusOfDtc;
744  }
745  else { /* Error while fetching the next DTC - do nothing */ }
746  }
747 
748  lub_retVal = E_OK;
749  }
750  else /* Cannot apply filter */
751  {
752  lub_retVal = E_NOT_OK;
753  }
754 
755  return lub_retVal;
756 }
