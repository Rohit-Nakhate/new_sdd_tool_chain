269  if ((TRUE == lub_KL15ON) && (lub_activeSession != DCM_DEFAULT_SESSION))

274

276  }
277  else { /* KL15 not changed or signal not received - do nothing */ }

279  else /* KL15 is off and reset to Default session is requested */

282  uint8 lub_index;
283 
284  for(lub_index = 0u; lub_index< DCM_NUM_MAIN_CONNECTIONS; lub_index++)
285  {

287  if ( Dcm_MainConnections[lub_index].Status != DCM_MAIN_CONNECTION_IDLE )
288  {
289  lub_doResetSession = FALSE;
290  break;
291  }
292  else { /* Do nothing */ }
293  }
294 
295  if ( FALSE != lub_doResetSession ) /* Request to Default session can be processed */
296  {
297  Std_ReturnType retVal;
298 
299  retVal = Dcm_ResetToDefaultSession();

305  rub_requestDefaultSession = FALSE;
306  }

308  }
