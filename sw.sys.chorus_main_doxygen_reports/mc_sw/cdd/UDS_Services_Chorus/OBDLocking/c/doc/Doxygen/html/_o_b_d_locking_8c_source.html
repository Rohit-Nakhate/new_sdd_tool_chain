262 /*this function process Routine0x0300, return TRUE if OBD Lock should be be deactivated*/
263 static boolean processRoutine0x0300(void)
264 {
265  boolean OBD_lock_Deactivated_by_0300 = FALSE;
266 
267  /* check timeout monitoring. */
268  if (Routine_0300_Timer > 0u)
269  {
270  /* The routine was started. We should send cyclically the OBD_locking_active flag as TRUE as long as ack signal is not received. */
271  boolean OBD_Unlock_R7_Ack = FALSE;
272  (void) Rte_Read_OBDLocking_OBDUnlockR7Ack_OBD_Unlock_R7_Ack (&OBD_Unlock_R7_Ack);
273 
274  if (OBD_Unlock_R7_Ack == FALSE)
275  {
276  /* ack not received. Send the OBD unlock info again */
277  (void) Rte_Write_OBDLocking_OBDUnlock_OBD_Unlock(TRUE);
278  Routine_0300_Timer --;
279  }
280  else
281  {
282  /* The Ack was received. We should stop the routine. Also we should stop the Diag routing locking here. */
283  Routine_0300_Timer = 0;
284  Routine_0300_State = FALSE;
285  (void) Rte_Write_OBDLocking_OBDUnlock_OBD_Unlock(FALSE);
286  OBD_lock_Deactivated_by_0300 = TRUE;
287  RC_FinishRoutine(RC_ROUTINE_INDEX_GW_Routine_0x0300);
288  GW_Routine_0x0300_SetInfo(RC_ROUTINE_RESULT_CORRECT_RESULTS);
289  }
290  }
291  else
292  {
293  /* timer has expired. Stop the routine with error. Should we stop the Diag routing locking? Better not. */
294  Routine_0300_State = FALSE;
295  (void) Rte_Write_OBDLocking_OBDUnlock_OBD_Unlock(FALSE);
296  (void) GW_Routine_0x0300_Abort();
297  }
298 
299  return OBD_lock_Deactivated_by_0300;
300 }

302 static void processOBDLocking(boolean deactivationRequestedBy0x0300)
303 {
304  static boolean s_kl_15_old = FALSE;
305  static boolean ORU_status_running_old = FALSE;
306  boolean s_kl_15 = FALSE;
307  uint8 ORU_status = ORU_STATUS_0;
308  boolean ORU_status_running = FALSE;
309 
310  (void)Rte_Read_OBDLocking_StORU_ORU_Status(&ORU_status);
311 
312  /* Check if the ORU is active */
313  if((ORU_status == ORU_STATUS_RUNNING) || (ORU_status == ORU_STATUS_RUNNING_HV))
314  {
315  ORU_status_running = TRUE;
316  }
317  /* if the ORU is inactive */
318  else
319  {
320  ORU_status_running = FALSE;
321  }
322 
323  (void)Rte_Read_OBDLocking_Kl15_s_Kl_15(&s_kl_15);
324 
325  /* Calculate the state of diag routing activation */
326  if (OBD_locking_active == FALSE)
327  {
328  /* check conditions for activating the OBD routing lock: ORU_status from IDLE to running */
329  if ((ORU_status_running == TRUE) && (ORU_status_running_old == FALSE))
330  {
331  OBD_locking_active = TRUE;
332  }
333  else
334  {
335  /* nothing to do */
336  }
337  }
338  else
339  {
340  /* check for conditions to deactivate the OBD_locking_active: ORU status, KL15 and 0300 routine */
341  if (((ORU_status_running == FALSE) && (ORU_status_running_old == TRUE)) ||
342  ((s_kl_15 == TRUE) && (s_kl_15_old == FALSE)) ||
343  (deactivationRequestedBy0x0300 == TRUE))
344  {
345  OBD_locking_active = FALSE;
346  }
347  else
348  {
349  /* Nothing to do */
350  }
351  }
352 
353  ORU_status_running_old = ORU_status_running;
354  s_kl_15_old = s_kl_15;
355 }
