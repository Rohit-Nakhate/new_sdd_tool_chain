917 static void ProcessVehicleSpeed(void)
918 {
919  /* TRACE_REQUIREMENT, SWR_ICAS1_Diag_118, 14/05/2020, KoRo, Complete */
920  static uint16 si_ESP_Signal = 0u;
921 
922  Std_ReturnType readStatus = Rte_Read_DiagnosticTimers_ESP21_u16_Speed(&si_ESP_Signal);
923 
924  if (E_NOT_OK != readStatus)
925  {
926  if ( ((uint16)K_DIAG_PRECOND_MIN_VEHICLE_SPEED >= si_ESP_Signal) || ((uint16)K_DIAG_PRECOND_MAX_VEHICLE_SPEED <= si_ESP_Signal)
927  || (RTE_E_MAX_AGE_EXCEEDED == readStatus) )
928  {
929  /* PASS if speed <= 5 km/h OR speed >= 65533 (undervoltage, Init, Error) OR Rte_Read return timeout value */
930  VehicleSpeedValidation.si_Status = PASSED;
931  VehicleSpeedValidation.TimerValue = (uint8)SPEED_VALIDATION_TIMER;
932  }
933  else if ( ((uint8)TIMER_ELAPSED_VALUE < VehicleSpeedValidation.TimerValue) )
934  {
935  /* Start and process 2s timer before speed will FAILED */
936  VehicleSpeedValidation.TimerValue--;
937  }
938  else
939  {
940  /* speed >= 500 AND speed < 65533 for longer than 2s */
941  VehicleSpeedValidation.si_Status = FAILED;
942  }
943  }
944  else
945  {
946  /* FAILED if signal was incorrectly readed from RTE */
947  VehicleSpeedValidation.si_Status = FAILED;
948  VehicleSpeedValidation.TimerValue = (uint8)SPEED_VALIDATION_TIMER;
