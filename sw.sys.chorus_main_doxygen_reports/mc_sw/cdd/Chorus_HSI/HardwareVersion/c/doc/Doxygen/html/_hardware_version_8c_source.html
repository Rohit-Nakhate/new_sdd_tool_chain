250 void HardwareVersion_GetHardwareVersion(HardwareVersionType* hwVersion)
251 {
252  HardwareVersion_ReadOtp();
253 
254  *hwVersion = NOT_DEFINED_HW;
255  uint32 HardwareVersion;
256 
257  /* The HW version identifier is stored in OTP flash at the address of VW_ECUHardwareVersionNumber_F1A3
258  * The scheme for HW version numbering is VersionPlatform | VersionHigh | VersionLow
259  * VersionPlatform is always 0 (0x30), VersionHigh is 1 for B0 (former name for B0 was A sample)
260  * For B samples VersionHigh is 2, for C sample Version high will be 3.
261  * VersionLow is incremented for each miner version of B or C sample.
262  *
263  * The identifiers are ASCII coded values. The following HW versions exist so far
264  * HW Name | PL HI LO | ID
265  * ---------------------------------
266  * B0 | 0x30, 0x31, 0x32 | 012
267  * B1 | 0x30, 0x32, 0x31 | 021
268  * B1.1 | 0x30, 0x32, 0x32 | 022
269  * B2 | 0x30, 0x32, 0x33 | 023
270  * B3 | 0x30, 0x32, 0x34 | 024
271  * C1 | 0x30, 0x33, 0x30 | 030
272  */
273 
274  /* Create an 32bit integer by combining HW version IDs */
275  HardwareVersion = (uint32)(HardwareVersionCopy.HardwareVersionPlatform) << 16u;
276  HardwareVersion |= (uint32)(HardwareVersionCopy.HardwareVersionHighAscii) << 8u;
277  HardwareVersion |= (uint32)(HardwareVersionCopy.HardwareVersionLowAscii);
278 
279  switch(HardwareVersion)
280  {
281  case 0x00593131u:
282  /* B1 Sample */
283  *hwVersion = VW_MEB19_EVO_B1_DEV;
284  break;
285  case 0x005A3131u:
286  /* B1.1 Sample */
287  /* The software does not differentiate between B1 and B1.1 sample. */
288  *hwVersion = VW_MEB19_EVO_B1_MAX_DEV;
289  break;
290  case 0x00593132u:
291  /* B2 Sample */
292  *hwVersion = VW_MEB19_EVO_B2_DEV;
293  break;
294  case 0x00593133u:
295  /* B2.TI Sample - also a simple B2.TI sample for the Chorus, but Chorus might have to forward this info to RCAR */
296  /* The software does not differentiate between B2 and B2.TI sample. */
297  *hwVersion = VW_MEB19_EVO_B2_TI_DEV;
298  break;
299  case 0x00593231u:
300  /* B2.1 Sample */
301  *hwVersion = VW_MEB19_EVO_C1_DEV;
302  break;
303  case 0x00303231u:
304  /* B2.1 Sample */
305  *hwVersion = VW_MEB19_EVO_C1_SERIAL;
306  break;
307  case 0x00593232u:
308  /* B2.1 Sample */
309  *hwVersion = VW_MEB19_EVO_Cx_DEV;
310  break;
311  case 0x00303232u:
312  /* B2.1 Sample */
313  *hwVersion = VW_MEB19_EVO_Cx_SERIAL;
314  break;
315  default:
316  /* Unknown HW version, should be later handled by FBL, so it does not start into APPL if no hardware version is given */
317  *hwVersion = NOT_DEFINED_HW;
318  break;
319  }
320 
321 }
