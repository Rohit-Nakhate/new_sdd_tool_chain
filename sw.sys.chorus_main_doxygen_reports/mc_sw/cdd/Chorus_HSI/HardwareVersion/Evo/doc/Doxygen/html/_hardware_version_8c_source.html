137 void HardwareVersion_GetHardwareVersion(HardwareVersionType* hwVersion)
138 {
139  HardwareVersion_ReadOtp();
140 
141  *hwVersion = NOT_DEFINED_HW;
142  uint32 HardwareVersion;
143 
144  /* The HW version identifier is stored in OTP flash at the address of VW_ECUHardwareVersionNumber_F1A3
145  * The scheme for HW version numbering is VersionPlatform | VersionHigh | VersionLow
146  * VersionPlatform is always 0 (0x30), VersionHigh is 1 for B0 (former name for B0 was A sample)
147  * For B samples VersionHigh is 2, for C sample Version high will be 3.
148  * VersionLow is incremented for each miner version of B or C sample.
149  *
150  * The identifiers are ASCII coded values. The following HW versions exist so far
151  * HW Name | PL HI LO | ID
152  * ---------------------------------
153  * B0 | 0x30, 0x31, 0x32 | 012
154  * B1 | 0x30, 0x32, 0x31 | 021
155  * B1.1 | 0x30, 0x32, 0x32 | 022
156  * B2 | 0x30, 0x32, 0x33 | 023
157  * B3 | 0x30, 0x32, 0x34 | 024
158  * C1 | 0x30, 0x33, 0x30 | 030
159  */
160 
161  /* Create an 32bit integer by combining HW version IDs */
162  HardwareVersion = (uint32)(HardwareVersionCopy.HardwareVersionPlatform) << 16u;
163  HardwareVersion |= (uint32)(HardwareVersionCopy.HardwareVersionHighAscii) << 8u;
164  HardwareVersion |= (uint32)(HardwareVersionCopy.HardwareVersionLowAscii);
165 
166  switch(HardwareVersion)
167  {
168  case 0x00593131u:
169  /* B1 Sample */
170  *hwVersion = VW_MEB19_EVO_B1_DEV;
171  break;
172  case 0x005A3131u:
173  /* B1.1 Sample */
174  /* The software does not differentiate between B1 and B1.1 sample. */
175  *hwVersion = VW_MEB19_EVO_B1_MAX_DEV;
176  break;
177  case 0x00593132u:
178  /* B2 Sample */
179  *hwVersion = VW_MEB19_EVO_B2_DEV;
180  break;
181  case 0x00593133u:
182  /* B2.TI Sample - also a simple B2.TI sample for the Chorus, but Chorus might have to forward this info to RCAR */
183  /* The software does not differentiate between B2 and B2.TI sample. */
184  *hwVersion = VW_MEB19_EVO_B2_TI_DEV;
185  break;
186  case 0x00593231u:
187  /* B2.1 Sample */
188  *hwVersion = VW_MEB19_EVO_C1_DEV;
189  break;
190  case 0x00303231u:
191  /* B2.1 Sample */
192  *hwVersion = VW_MEB19_EVO_C1_SERIAL;
193  break;
194  case 0x00593232u:
195  /* B2.1 Sample */
196  *hwVersion = VW_MEB19_EVO_Cx_DEV;
197  break;
198  case 0x00303232u:
199  /* B2.1 Sample */
200  *hwVersion = VW_MEB19_EVO_Cx_SERIAL;
201  break;
202  default:
203  /* Unknown HW version, should be later handled by FBL, so it does not start into APPL if no hardware version is given */
204  *hwVersion = NOT_DEFINED_HW;
205  break;
206  }
207 
208 }
