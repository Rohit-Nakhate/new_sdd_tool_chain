180  uint8 ui8_VoltageStatus = 0u;
181 
182  Std_ReturnType retVal = E_OK;
183 
184  /* check if voltage measurement was initialized */

187  /* Gets voltage value in mV and status */
188  ui8_VoltageStatus = VoltageMeasurement_Chorus_KL30_mV(&Voltage);
189 
190  /* is voltage measurement valid */
191  if (TRUE == ui8_VoltageStatus)
192  {
193  if ((uint16)ERR_UNDER_VOLTAGE_THRESHOLD > Voltage)
194  {
195  /* Battery-Voltage is below under-voltage-threshold */
196  // under-voltage is present
197  ErrData.VoltageRange = E_UNDER_VOLTAGE;
198  }
199  /* SWATT-coverage-off FALSE == (ERR_UNDER_VOLTAGE_THRESHOLD <= Voltage) executed in previous condition */
200  else if ((uint16)ERR_LOWER_NORMAL_VOLTAGE_THRESHOLD > Voltage)
201  /* SWATT-coverage-on */
202  {
203  /* Battery-Voltage is between under-voltage-hysteresis-thresholds */
204  if (E_UNDER_VOLTAGE != ErrData.VoltageRange)
205  {
206  ErrData.VoltageRange = E_NORMAL_VOLTAGE;
207  }
208  }
209  /* SWATT-coverage-off First condition fail already checked */
210  else if ((uint16)ERR_UPPER_NORMAL_VOLTAGE_THRESHOLD >= Voltage)
211  /* SWATT-coverage-on */
212  {
213  /* Battery-Voltage is in normal voltage area */
214  ErrData.VoltageRange = E_NORMAL_VOLTAGE;
215 
216  }
217  /* SWATT-coverage-off First condition fail already checked */
218  else if ((uint16)ERR_OVER_VOLTAGE_THRESHOLD >= Voltage)
219  /* SWATT-coverage-on */
220  {
221  /* Battery-Voltage is between over-voltage-hysteresis-thresholds */
222  if (E_OVER_VOLTAGE != ErrData.VoltageRange)
