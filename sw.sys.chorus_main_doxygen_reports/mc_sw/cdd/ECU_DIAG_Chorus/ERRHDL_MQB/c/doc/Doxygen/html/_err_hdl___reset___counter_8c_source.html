287 static void ErrProcessResetCounters(void)
288 {
289  Std_ReturnType readRetVal = E_OK;
290  uint16 dataCountReset = 0x00u;
291  uint8 resetMask = 0x00u;
292  uint8 Index = 0x00u;
293 
294  static const MapRstPtrrToFun mapRstPtrrToFun[NumberOfResetCounterFunctions] =
295  {
296  {Get_SysInt_Wdg_Reset_Delta_Ctr_u16, ExternalWatchdogResets},
297  {Get_SysInt_Functional_Reset_Delta_Ctr_u16, FunctionalResets},
298  {Get_SysInt_Destructive_Reset_Delta_Ctr_u16, DestructiveResets}
299  };
300 
301  readRetVal |= Get_SysInt_ResetType_Bitmask_u8(&resetMask);
302 
303  for(Index = 0x00u; Index < (uint8)NumberOfResetCounterFunctions; Index++)
304  {
305  if((E_NOT_OK != readRetVal) && (TRUE == (uint8)(((uint16)resetMask >> (uint16)Index) & (uint16)0x01u)))
306  {
307  readRetVal |= (*mapRstPtrrToFun[Index].pointerToResetCountFunction)(&dataCountReset);
308 
309  if(E_NOT_OK != readRetVal)
310  {
311  ErrAddResetCounters(mapRstPtrrToFun[Index].resetCounterSource, dataCountReset);
312  }
313  else{/*MISRA*/}
314  }
315  else{/*MISRA*/}
316  }
317 
318  if(E_NOT_OK != readRetVal)
319  {
320  ErrWriteNVMResetCounters(NVM_PENDING);
321  }
322  else{/*MISRA*/}

331 static void ErrAddResetCounters(const ResetCounterSources resetCounterSource, const uint16 dataCountReset)
332 {
333  /* This Switch case adds new reset to the counters*/
334  switch(resetCounterSource)
335  {
336  case ExternalWatchdogResets:
337  {
338  /* This counter will count for external watchdog resets*/
339  GW_Unexpected_Resets_Counter.NumberExternalWatchdogResets += dataCountReset;
340  break;
341  }
342  case FunctionalResets:
343  {
344  /* This counter will count for funtional resets*/
345  GW_Unexpected_Resets_Counter.NumberFunctionalResets += dataCountReset;
346  break;
347  }
348  case DestructiveResets:
349  {
350  /* This counter will count for destructive resets*/
351  GW_Unexpected_Resets_Counter.NumberDestructiveResets += dataCountReset;
352  break;
353  }
354  default:
355  {
356  /*Default case - Invalid value*/
357  break;
358  }
