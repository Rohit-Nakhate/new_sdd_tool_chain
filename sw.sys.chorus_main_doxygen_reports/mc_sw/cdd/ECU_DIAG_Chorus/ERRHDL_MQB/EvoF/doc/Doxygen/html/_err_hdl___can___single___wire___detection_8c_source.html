408 static boolean ErrHdlIsSingleWireErrorPresent(const TComChannel Channel)
409 {
410  uint16 ui16_CanHighValue = 0x00u;
411  uint16 ui16_CanLowValue = 0x00u;
412  uint32 temp = 0x00u;
413  uint16 diff = 0x00u;
414  boolean retVal = FALSE;
415  uint32 ecf = 0x00u;
416 
417  ecf = (uint32)hal_GetErrorCorrectionFactor();
418 
419  /* read CanLow and CanHigh voltage values*/
420  HalReadCanVoltage(Channel, &ui16_CanHighValue, &ui16_CanLowValue);
421 
422  /* Correct the values with the Error Correction Factor */
423  temp = ((uint32)((uint32)ui16_CanHighValue*(uint32)ecf))/(uint32)HAL_GET_ERROR_CORRECT_IONFACTOR;
424  ui16_CanHighValue = (uint16)temp;
425  temp = ((uint32)((uint32)ui16_CanLowValue*(uint32)ecf))/(uint32)HAL_GET_ERROR_CORRECT_IONFACTOR;
426  ui16_CanLowValue = (uint16)temp;
427  if (ui16_CanHighValue >= ui16_CanLowValue)
428  {
429  diff = ui16_CanHighValue - ui16_CanLowValue;
430  }
431  else
432  {
433  diff = ui16_CanLowValue - ui16_CanHighValue;
434  }
435  /* SWATT-coverage-off Can't be covered because of if logic */
436  /* (sw_diff_AI_CANX <= 0,1V) AND (sw_AI_CANX_L >= 3,2V) AND (sw_AI_CANX_H >= 3,2V) */
437  /* (sw_diff_AI_CANX <= 0,3V) AND (sw_AI_CANX_L <= 0,3V) AND (sw_AI_CANX_H <= 0,3V) */
438  /* (sw_diff_AI_CANX <= 1,55V) AND (sw_AI_CANX_L <= 0,4V) AND (sw_AI_CANX_H <= 1,55V) */
439  if (((diff <= (uint16)CAN_DIFF_0_1V) && (ui16_CanLowValue >= (uint16)CANL_SHORT_TO_UBATT_3_2V) && (ui16_CanHighValue >= (uint16)CANH_SHORT_TO_UBATT_3_2V)) ||
440  ((diff <= (uint16)CAN_DIFF_0_3V) && (ui16_CanLowValue <= (uint16)CANL_SHORT_TO_GND_0_3V) && (ui16_CanHighValue <= (uint16)CANH_SHORT_TO_GND_0_3V)) ||
441  ((diff <= (uint16)CAN_DIFF_1_55V) && (ui16_CanLowValue <= (uint16)CANL_SHORT_0_4V) && (ui16_CanHighValue <= (uint16)CANH_SHORT_1_55V)))
442  {
443  retVal = TRUE;
444  }
445  /* SWATT-coverage-on */
446 
447  return retVal;
