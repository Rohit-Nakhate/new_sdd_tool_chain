248 boolean ErrRemoteCanGetBusStateFromR7(const uint8 Channel)
249 {
250  static const tNmhStateType R7_to_Ch_NM_State[7]=
251  {
252  kNmhStateSleep, // NM_STATE_UNINIT = 0
253  kNmhStateSleep, // NM_STATE_BUS_SLEEP = 1
254  kNmhStatePrepareSleep, // NM_STATE_PREPARE_BUS_SLEEP = 2
255  kNmhStateReadySleep, // NM_STATE_READY_SLEEP = 3
256  kNmhStateNormalMode, // NM_STATE_NORMAL_OPERATION = 4
257  kNmhStateRepeatMsg, // NM_STATE_REPEAT_MESSAGE = 5
258  kNmhStateRepeatMsg // NM_STATE_SYNCHRONIZE = 6
259  };
260 
261  boolean BusState = FALSE;
262  uint8 Nm_State = 0x00u;
263 
264  static const PtrToRemoteCanInt RemoteCanStates[ERR_NUMB_OF_REMOTE_BUSSES] =
265  {
266  Rte_Read_ERRHDL_MQB_InPort_s_CAN_AFS_NM_state,
267  Rte_Read_ERRHDL_MQB_InPort_s_CAN_MFL_NM_state
268  };
269 
270  Std_ReturnType ReturnValue = (*RemoteCanStates[REMOTE_CAN_IDX(Channel)])(&Nm_State);
271 
272  if (E_OK == ReturnValue)
273  {
274  if (Nm_State < 7U)
275  {
276  if((tNmhStateType)kNmhStateReadySleep >= (tNmhStateType)R7_to_Ch_NM_State[Nm_State])
277  {
278  BusState = TRUE;

280  else
281  {

283  }
