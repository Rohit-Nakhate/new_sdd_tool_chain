226 void ErrSteuergeraetErrDTCMonitoring(void)

236 
237  /* To check if DTC need to be handled */
238  if(FALSE == isSysIntDTCHandled)
239  {
240  /* Check if the System Integrity is initialized or not */
241  if(TRUE == SystemIntegrity_SysIntIsInitialized())
242  {
243  /* Get the system DTC Key */
244  ReturnValue = Get_SysInt_DTC_Key(&SysInitDTCKey);
245  ReturnValue |= Get_SysInt_DTC_Key_Inverted(&InvSysInitDTCKey);
246  /* If no DTC key is detected */
247  if((E_OK == ReturnValue) && (NO_DTC_KEY == SysInitDTCKey) && (NO_DTC_KEY_INVERTED == InvSysInitDTCKey))
248  {
249  if(CurrentHandledDTC < (uint8)NUMBER_Of_Steuergeraet_DTC)
250  {
251  /* Get the Error masking value */
252  ReturnValue |= ErrSetSteuergeraetErrBuffer();
253 
254  /* Check if the DTC mask id proper */
255  if((E_OK == ReturnValue) && (MASK_IS_EMPTY < DTCResonT[CurrentHandledDTC]) && (FALSE == DTCCanBeSet))
256  {
257  /* Check if the voltage is in range */
258  ReturnValue |= ErrIsVoltageOutOfRange(&isVoltageOutOfRange);

260  /* If voltage is in range */
261  if((E_OK == ReturnValue) && (FALSE == isVoltageOutOfRange))
262  {
263  /* Set the DTC */
264  ErrReasonBuffer = DTCResonT[CurrentHandledDTC];
265  DTCCanBeSet = TRUE;
266  Dem_ReportErrorStatus(SteuergeraetErrDTCt[CurrentHandledDTC], DEM_EVENT_STATUS_FAILED);
267  }
268  else {/* Do nothing */}
269  }
270  /* No error. Then check for the next DTC. */
271  else if (FALSE == DTCCanBeSet)

273  ErrGotToNextDTC();
274  }
275  else {/* Do nothing */}
276  }
