78 typedef enum
79 {
80  /* For all routines from default group (RC_GROUP_DEFAULT) */
81  ROUTINE_STATUS_INACTIVE = 0x00u, /* If the routine has not been executed in current operation cycle */
82  ROUTINE_STATUS_ACTIVE = 0x01u, /* If the StartRoutine request is received, and the GW starts the execution of the routine */
83  ROUTINE_STATUS_ABORTED = 0x02u, /* If the routine is stopped because of an error or safety reason, or a session change to default session */
84  ROUTINE_STATUS_EXECUTION_STUCK = 0x03u, /* Routine is being executed, but cannot be stopped */
85  ROUTINE_STATUS_FINISHED_CORRECTLY = 0x04u, /* The routine has been successfully executed in the current operation cycle */
86  ROUTINE_STATUS_RETURNED_TO_ECU_CONTROL = 0x05u, /* The routine has been successfully stopped via StopRoutine request */
87  ROUTINE_STATUS_SYSTEM_SUPPLIER_DEFINED_REACHED_STATUS = 0x06u, /* Reserved Status for specific purposes - if a different status is */
88  /* needed for a routine, this will be defined in the corresponding requirement for the routine */
89 
90  /* For all routines from extended group (RC_GROUP_EXTENDED) */
91  EXTENDED_ROUTINE_STATUS_ACTIVE = 0x01u,
92  EXTENDED_ROUTINE_STATUS_NOT_ACTIVE_OR_FINISHED_CORRECTLY = 0x02u,
